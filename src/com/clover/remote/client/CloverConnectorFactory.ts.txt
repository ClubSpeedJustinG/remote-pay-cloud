The implementation of factories can be done in later libraries or applications that require this one.
That will allow the implementation of specific objects (like websockets) to be delayed until they are better known.

EX:  A browser implementation would not need to require a WebSSocket impl, whereas a NodeJS impl would.

import sdk = require('remote-pay-cloud-api');
import {CloverDeviceConfiguration} from './CloverDeviceConfiguration';

/**
 * Creates a ICloverConnector for a given configuration.
 *
 */
export class CloverConnectorFactory {
    public static createICloverConnector(configuration: CloverDeviceConfiguration):sdk.remotepay.ICloverConnector {
        return new Clover.CloverConnector(configuration);
    }
}
